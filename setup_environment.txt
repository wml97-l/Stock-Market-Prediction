#!/bin/bash

# --- Configuration ---
ENV_NAME="lecturer_notebook_env"
KERNEL_DISPLAY_NAME="Lecturer PySpark Env"

echo "--- Jupyter Environment Setup Script ---"

# 1. Check for Python 3
if ! command -v python3 &> /dev/null
then
    echo "Error: python3 could not be found. Please ensure Python 3 is installed."
    exit 1
fi

# 2. Check for the required packages file
if [ ! -f "requirements.txt" ]; then
    echo "Error: 'requirements.txt' not found in the current directory."
    echo "Please run 'pip freeze > requirements.txt' in your environment first."
    exit 1
fi

# 3. Create a new virtual environment
echo "Creating virtual environment: $ENV_NAME"
python3 -m venv "$ENV_NAME"

# 4. Activate the virtual environment
source "$ENV_NAME/bin/activate"

echo "Virtual environment activated."

# 5. Install the required packages (ipykernel first)
echo "Installing ipykernel and updating pip..."
pip install --upgrade pip
pip install ipykernel

# 6. Install all dependencies from the requirements file
echo "Installing dependencies from requirements.txt..."
# Use -q (quiet) to suppress lengthy output during installation
pip install -r requirements.txt -q

# 7. Register the new environment as a Jupyter kernel
echo "Registering environment as a Jupyter kernel..."
python3 -m ipykernel install --user --name="$ENV_NAME" --display-name "$KERNEL_DISPLAY_NAME"

# 8. Deactivate the environment
deactivate

echo "==================================================="
echo "SETUP COMPLETE!"
echo "==================================================="
echo "To run the notebook:"
echo "1. Start your Jupyter server (e.g., 'jupyter notebook')."
echo "2. Open the notebook file."
echo "3. Go to the 'Kernel' menu and select '$KERNEL_DISPLAY_NAME'."
